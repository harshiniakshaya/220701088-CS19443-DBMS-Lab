SYNTAX: SIMPLE IF-ELSIF-ELSE
declare 
  variable datatype;
begin
  variable:=&variable; //to get input from the user
  if(condition) then
    statements;
  elsif(condition) then
    statements;
  else
    statements;
  end if;
end;
/
